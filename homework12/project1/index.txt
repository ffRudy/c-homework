<!DOCTYPE html>
<html>
<head>
    <title>Order App</title>
</head>
<body>

    <div>
        <h2>All Orders</h2>
        <ul id="orders" />
    </div>
    <div>
        <h2>Search by ID</h2>
        <input type="text" id="OrderId" size="5" />
        <input type="button" value="Search" onclick="find();" />
        <p id="order" />
    </div>

    <script src="http://ajax.aspnetcdn.com/ajax/jQuery/jquery-2.0.3.min.js"></script>
    <script>
        var uri = 'api/orders';

        $(document).ready(function () {
            // Send an AJAX request
            $.getJSON(uri)
                .done(function (data) {
                    // On success, 'data' contains a list of orders.
                    $.each(data, function (key, item) {
                        // Add a list item for the order.
                        $('<li>', { text: formatItem(item) }).appendTo($('#orders'));
                    });
                });
        });

        function formatItem(item) {
            return item.Customer +","+ item.OrderId;
        }

        function find() {
            var id = $('#OrderId').val();
            $.getJSON(uri + '/' + id)
                .done(function (data) {
                    $('#order').text(formatItem(data));
                })
                .fail(function (jqXHR, textStatus, err) {
                    $('#order').text('Error: ' + err);
                });
        }
    </script>
</body>
</html>
